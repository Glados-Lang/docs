```rust
fn is_prime(nb: u32) -> bool
{
    let mut i: u32 = 2;

    if (nb == 0 || nb == 1) {
        return false;
    }
    while (i < nb) {
        if (nb % i == 0) {
            return false;
        }
        i += 1;
    }
    return true;
}
```
